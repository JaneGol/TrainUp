üü¢  AGREED FORMULA  ‚Äì  (confirmed with S&C coach)
-------------------------------------------------
session_load (AU) =
    RPE √ó emotional_load (1‚Äì1.5) √ó duration_minutes √ó coeff(training_type)

coeff:
  Field  = 1.2
  Gym    = 1.0
  Match  = 1.5

When two Field sessions occur on the same calendar day, **AU is the sum of both
session_load values**. All downstream aggregates (daily, weekly, ACWR) Sum‚ÜíThen‚ÜíCalc.
No averages except ACWR.

        WHEN 'Gym'   THEN 1.0
        ELSE 1.5
      END);

CREATE OR REPLACE FUNCTION calc_session_load() RETURNS TRIGGER AS $$
BEGIN
  NEW.session_load :=
       ROUND( NEW.rpe * NEW.emotional_load * NEW.duration_minutes *
              CASE NEW.type
                WHEN 'Field' THEN 1.2
                WHEN 'Gym'   THEN 1.0
                ELSE 1.5
              END);
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_calc_session_load
BEFORE INSERT OR UPDATE ON training_sessions
FOR EACH ROW EXECUTE FUNCTION calc_session_load();
